67 GUIJunctionWrapper::GUIJunctionWrapper(MSJunction& junction)
68  : GUIGlObject(GLO_JUNCTION, junction.getID()),
69  myJunction(junction) {
70  if (myJunction.getShape().size() == 0) {
71  Position pos = myJunction.getPosition();
72  myBoundary = Boundary(pos.x() - 1., pos.y() - 1., pos.x() + 1., pos.y() + 1.);
73  } else {
74  myBoundary = myJunction.getShape().getBoxBoundary();
75  }
76  myMaxSize = MAX2(myBoundary.getWidth(), myBoundary.getHeight());
77 #ifdef HAVE_INTERNAL_LANES
78  myIsInner = dynamic_cast<MSInternalJunction*>(&myJunction) != 0;
79 #else
80  myIsInner = false;
81 #endif
82  myAmWaterway = myJunction.getIncoming().size() + myJunction.getOutgoing().size() > 0;
83  for (ConstMSEdgeVector::const_iterator it = myJunction.getIncoming().begin(); it != myJunction.getIncoming().end(); ++it) {
84  if (!(*it)->isInternal() && !isWaterway((*it)->getPermissions())) {
85  myAmWaterway = false;
86  break;
87  }
88  }
89  for (ConstMSEdgeVector::const_iterator it = myJunction.getOutgoing().begin(); it != myJunction.getOutgoing().end(); ++it) {
90  if (!(*it)->isInternal() && !isWaterway((*it)->getPermissions())) {
91  myAmWaterway = false;
92  break;
93  }
94  }
95 }

129 void
130 GUIJunctionWrapper::drawGL(const GUIVisualizationSettings& s) const {
131  // check whether it is not too small
132  if (s.scale * myMaxSize < 1.) {
133  return;
134  }
135  if (!myIsInner && s.drawJunctionShape) {
136  glPushMatrix();
137  glPushName(getGlID());
138  const SUMOReal colorValue = getColorValue(s);
139  GLHelper::setColor(s.junctionColorer.getScheme().getColor(colorValue));
140 
141  // recognize full transparency and simply don't draw
142  GLfloat color[4];
143  glGetFloatv(GL_CURRENT_COLOR, color);
144  const SUMOReal exaggeration = s.junctionSize.getExaggeration(s);
145  if (color[3] != 0 && s.scale * exaggeration > s.junctionSize.minSize) {
146  PositionVector shape = myJunction.getShape();
147  shape.closePolygon();
148  if (exaggeration > 1) {
149  shape.scaleRelative(exaggeration);
150  }
151  glTranslated(0, 0, getType());
152  if (s.scale * myMaxSize < 40.) {
153  GLHelper::drawFilledPoly(shape, true);
154  } else {
155  GLHelper::drawFilledPolyTesselated(shape, true);
156  }
157 #ifdef GUIJunctionWrapper_DEBUG_DRAW_NODE_SHAPE_VERTICES
158  GLHelper::debugVertices(shape, 80 / s.scale);
159 #endif
160  }
161  glPopName();
162  glPopMatrix();
163  }
164  if (myIsInner) {
165  drawName(myJunction.getPosition(), s.scale, s.internalJunctionName);
166  } else {
167  drawName(myJunction.getPosition(), s.scale, s.junctionName);
168  }
169 }

172 SUMOReal
173 GUIJunctionWrapper::getColorValue(const GUIVisualizationSettings& s) const {
174  switch (s.junctionColorer.getActive()) {
175  case 0:
176  if (myAmWaterway) {
177  return 1;
178  } else {
179  return 0;
180  }
181  case 1:
182  return gSelected.isSelected(getType(), getGlID()) ? 1 : 0;
183  case 2:
184  switch (myJunction.getType()) {
185  case NODETYPE_TRAFFIC_LIGHT:
186  return 0;
187  case NODETYPE_TRAFFIC_LIGHT_NOJUNCTION:
188  return 1;
189  case NODETYPE_PRIORITY:
190  return 2;
191  case NODETYPE_PRIORITY_STOP:
192  return 3;
193  case NODETYPE_RIGHT_BEFORE_LEFT:
194  return 4;
195  case NODETYPE_ALLWAY_STOP:
196  return 5;
197  case NODETYPE_DISTRICT:
198  return 6;
199  case NODETYPE_NOJUNCTION:
200  return 7;
201  case NODETYPE_DEAD_END:
202  case NODETYPE_DEAD_END_DEPRECATED:
203  return 8;
204  case NODETYPE_UNKNOWN:
205  case NODETYPE_INTERNAL:
206  assert(false);
207  return 8;
208  case NODETYPE_RAIL_SIGNAL:
209  return 9;
210  case NODETYPE_ZIPPER:
211  return 10;
212  case NODETYPE_TRAFFIC_LIGHT_RIGHT_ON_RED:
213  return 11;
214  case NODETYPE_RAIL_CROSSING:
215  return 12;
216  }
217  default:
218  assert(false);
219  return 0;
220  }
221 }
