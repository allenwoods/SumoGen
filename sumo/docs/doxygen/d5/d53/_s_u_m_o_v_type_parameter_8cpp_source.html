197 void
198 SUMOVTypeParameter::write(OutputDevice& dev) const {
199  if (onlyReferenced) {
200  return;
201  }
202  dev.openTag(SUMO_TAG_VTYPE);
203  dev.writeAttr(SUMO_ATTR_ID, id);
204  if (wasSet(VTYPEPARS_LENGTH_SET)) {
205  dev.writeAttr(SUMO_ATTR_LENGTH, length);
206  }
207  if (wasSet(VTYPEPARS_MINGAP_SET)) {
208  dev.writeAttr(SUMO_ATTR_MINGAP, minGap);
209  }
210  if (wasSet(VTYPEPARS_MAXSPEED_SET)) {
211  dev.writeAttr(SUMO_ATTR_MAXSPEED, maxSpeed);
212  }
213  if (wasSet(VTYPEPARS_PROBABILITY_SET)) {
214  dev.writeAttr(SUMO_ATTR_PROB, defaultProbability);
215  }
216  if (wasSet(VTYPEPARS_SPEEDFACTOR_SET)) {
217  dev.writeAttr(SUMO_ATTR_SPEEDFACTOR, speedFactor);
218  }
219  if (wasSet(VTYPEPARS_SPEEDDEVIATION_SET)) {
220  dev.writeAttr(SUMO_ATTR_SPEEDDEV, speedDev);
221  }
222  if (wasSet(VTYPEPARS_VEHICLECLASS_SET)) {
223  dev.writeAttr(SUMO_ATTR_VCLASS, toString(vehicleClass));
224  }
225  if (wasSet(VTYPEPARS_EMISSIONCLASS_SET)) {
226  dev.writeAttr(SUMO_ATTR_EMISSIONCLASS, PollutantsInterface::getName(emissionClass));
227  }
228  if (wasSet(VTYPEPARS_IMPATIENCE_SET)) {
229  if (impatience == -std::numeric_limits<SUMOReal>::max()) {
230  dev.writeAttr(SUMO_ATTR_IMPATIENCE, "off");
231  } else {
232  dev.writeAttr(SUMO_ATTR_IMPATIENCE, impatience);
233  }
234  }
235  if (wasSet(VTYPEPARS_SHAPE_SET)) {
236  dev.writeAttr(SUMO_ATTR_GUISHAPE, getVehicleShapeName(shape));
237  }
238  if (wasSet(VTYPEPARS_WIDTH_SET)) {
239  dev.writeAttr(SUMO_ATTR_WIDTH, width);
240  }
241  if (wasSet(VTYPEPARS_HEIGHT_SET)) {
242  dev.writeAttr(SUMO_ATTR_HEIGHT, height);
243  }
244  if (wasSet(VTYPEPARS_COLOR_SET)) {
245  dev.writeAttr(SUMO_ATTR_COLOR, color);
246  }
247  if (wasSet(VTYPEPARS_OSGFILE_SET)) {
248  dev.writeAttr(SUMO_ATTR_OSGFILE, osgFile);
249  }
250  if (wasSet(VTYPEPARS_IMGFILE_SET)) {
251  dev.writeAttr(SUMO_ATTR_IMGFILE, imgFile);
252  }
253  if (wasSet(VTYPEPARS_PERSON_CAPACITY)) {
254  dev.writeAttr(SUMO_ATTR_PERSON_CAPACITY, personCapacity);
255  }
256  if (wasSet(VTYPEPARS_CONTAINER_CAPACITY)) {
257  dev.writeAttr(SUMO_ATTR_CONTAINER_CAPACITY, containerCapacity);
258  }
259  if (wasSet(VTYPEPARS_BOARDING_DURATION)) {
260  dev.writeAttr(SUMO_ATTR_BOARDING_DURATION, boardingDuration);
261  }
262  if (wasSet(VTYPEPARS_LOADING_DURATION)) {
263  dev.writeAttr(SUMO_ATTR_LOADING_DURATION, loadingDuration);
264  }
265  if (wasSet(VTYPEPARS_MAXSPEED_LAT_SET)) {
266  dev.writeAttr(SUMO_ATTR_MAXSPEED_LAT, maxSpeedLat);
267  }
268  if (wasSet(VTYPEPARS_LATALIGNMENT_SET)) {
269  dev.writeAttr(SUMO_ATTR_LATALIGNMENT, latAlignment);
270  }
271  if (wasSet(VTYPEPARS_MINGAP_LAT_SET)) {
272  dev.writeAttr(SUMO_ATTR_MINGAP_LAT, minGapLat);
273  }
274  if (wasSet(VTYPEPARS_LANE_CHANGE_MODEL_SET) || lcParameter.size() != 0) {
275  dev.writeAttr(SUMO_ATTR_LANE_CHANGE_MODEL, lcModel);
276  std::vector<SumoXMLAttr> attrs;
277  for (SubParams::const_iterator i = lcParameter.begin(); i != lcParameter.end(); ++i) {
278  attrs.push_back(i->first);
279  }
280  std::sort(attrs.begin(), attrs.end());
281  for (std::vector<SumoXMLAttr>::const_iterator i = attrs.begin(); i != attrs.end(); ++i) {
282  dev.writeAttr(*i, lcParameter.find(*i)->second);
283  }
284  }
285  if (wasSet(VTYPEPARS_CAR_FOLLOW_MODEL) || cfParameter.size() != 0) {
286  dev.openTag(cfModel);
287  std::vector<SumoXMLAttr> attrs;
288  for (SubParams::const_iterator i = cfParameter.begin(); i != cfParameter.end(); ++i) {
289  attrs.push_back(i->first);
290  }
291  std::sort(attrs.begin(), attrs.end());
292  for (std::vector<SumoXMLAttr>::const_iterator i = attrs.begin(); i != attrs.end(); ++i) {
293  dev.writeAttr(*i, cfParameter.find(*i)->second);
294  }
295  dev.closeTag();
296  }
297  for (std::map<std::string, std::string>::const_iterator j = getMap().begin(); j != getMap().end(); ++j) {
298  dev.openTag(SUMO_TAG_PARAM);
299  dev.writeAttr(SUMO_ATTR_KEY, (*j).first);
300  dev.writeAttr(SUMO_ATTR_VALUE, (*j).second);
301  dev.closeTag();
302  }
303  dev.closeTag();
304 }
