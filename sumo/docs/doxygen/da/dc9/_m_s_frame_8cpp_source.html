444 bool
445 MSFrame::checkOptions() {
446  OptionsCont& oc = OptionsCont::getOptions();
447  bool ok = true;
448  if (!oc.isSet("net-file")) {
449  WRITE_ERROR("No network file (-n) specified.");
450  ok = false;
