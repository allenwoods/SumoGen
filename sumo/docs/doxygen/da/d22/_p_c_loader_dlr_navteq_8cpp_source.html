108 void
109 PCLoaderDlrNavteq::loadPOIFile(const std::string& file,
110  OptionsCont& oc, PCPolyContainer& toFill,
111  PCTypeMap& tm) {
112  // get the defaults
113  RGBColor c = RGBColor::parseColor(oc.getString("color"));
114  // parse
115  int l = 0;
116  LineReader lr(file);
117  while (lr.hasMore()) {
118  std::string line = lr.readLine();
119  ++l;
120  // skip invalid/empty lines
121  if (line.length() == 0 || line.find("#") != std::string::npos) {
122  continue;
123  }
124  if (StringUtils::prune(line) == "") {
125  continue;
126  }
127  // parse the poi
128  std::istringstream stream(line);
129  // attributes of the poi
130  std::string name, skip, type, desc;
131  std::getline(stream, name, '\t');
132  std::getline(stream, skip, '\t');
133  std::getline(stream, type, '\t');
134  std::getline(stream, desc, '\t');
135  if (stream.fail()) {
136  throw ProcessError("Invalid dlr-navteq-poi in line " + toString(l) + ":\n" + line);
137  }
138  double x, y;
139  stream >> x;
140  if (stream.fail()) {
141  throw ProcessError("Invalid x coordinate for POI '" + name + "'.");
142  }
143  stream >> y;
144  if (stream.fail()) {
145  throw ProcessError("Invalid y coordinate for POI '" + name + "'.");
146  }
147  Position pos(x, y);
148  // check the poi
149  if (name == "") {
150  throw ProcessError("The name of a POI is missing.");
151  }
152  if (!GeoConvHelper::getProcessing().x2cartesian(pos, true)) {
153  throw ProcessError("Unable to project coordinates for POI '" + name + "'.");
154  }
155 
156  // patch the values
157  bool discard = oc.getBool("discard");
158  SUMOReal layer = oc.getFloat("layer");
159  RGBColor color;
160  if (tm.has(type)) {
161  const PCTypeMap::TypeDef& def = tm.get(type);
162  name = def.prefix + name;
163  type = def.id;
164  color = def.color;
165  discard = def.discard;
166  layer = def.layer;
167  } else {
168  name = oc.getString("prefix") + name;
169  type = oc.getString("type");
170  color = c;
171  }
172  if (!discard) {
173  PointOfInterest* poi = new PointOfInterest(name, type, color, pos, layer);
174  toFill.add(poi, OptionsCont::getOptions().isInStringVector("prune.keep-list", name));
175  }
176  }
177 }

180 void
181 PCLoaderDlrNavteq::loadPolyFile(const std::string& file,
182  OptionsCont& oc, PCPolyContainer& toFill,
183  PCTypeMap& tm) {
184  // get the defaults
185  RGBColor c = RGBColor::parseColor(oc.getString("color"));
186  // attributes of the poly
187  // parse
188  int l = 0;
189  LineReader lr(file);
190  while (lr.hasMore()) {
191  std::string line = lr.readLine();
192  ++l;
193  // skip invalid/empty lines
194  if (line.length() == 0 || line.find("#") != std::string::npos) {
195  continue;
196  }
197  if (StringUtils::prune(line) == "") {
198  continue;
199  }
200  // parse the poi
201  StringTokenizer st(line, "\t");
202  std::vector<std::string> values = st.getVector();
203  if (values.size() < 6 || values.size() % 2 != 0) {
204  throw ProcessError("Invalid dlr-navteq-polygon - line: '" + line + "'.");
205  }
206  std::string id = values[0];
207  std::string ort = values[1];
208  std::string type = values[2];
209  std::string name = values[3];
210  PositionVector vec;
211  int index = 4;
212  // now collect the positions
213  while ((int)values.size() > index) {
214  std::string xpos = values[index];
215  std::string ypos = values[index + 1];
216  index += 2;
217  SUMOReal x = TplConvert::_2SUMOReal(xpos.c_str());
218  SUMOReal y = TplConvert::_2SUMOReal(ypos.c_str());
219  Position pos(x, y);
220  if (!GeoConvHelper::getProcessing().x2cartesian(pos)) {
221  WRITE_WARNING("Unable to project coordinates for polygon '" + id + "'.");
222  }
223  vec.push_back(pos);
224  }
225 
226  name = StringUtils::convertUmlaute(name);
227  if (name == "noname" || toFill.getPolygons().get(name) != 0) {
228  name = name + "#" + toString(toFill.getEnumIDFor(name));
229  }
230 
231  // check the polygon
232  if (vec.size() == 0) {
233  WRITE_WARNING("The polygon '" + id + "' is empty.");
234  continue;
235  }
236  if (id == "") {
237  WRITE_WARNING("The name of a polygon is missing; it will be discarded.");
238  continue;
239  }
240 
241  // patch the values
242  bool fill = vec.front() == vec.back();
243  bool discard = oc.getBool("discard");
244  SUMOReal layer = oc.getFloat("layer");
245  RGBColor color;
246  if (tm.has(type)) {
247  const PCTypeMap::TypeDef& def = tm.get(type);
248  name = def.prefix + name;
249  type = def.id;
250  color = def.color;
251  fill = fill && def.allowFill;
252  discard = def.discard;
253  layer = def.layer;
254  } else {
255  name = oc.getString("prefix") + name;
256  type = oc.getString("type");
257  color = c;
258  }
259  if (!discard) {
260  SUMO::Polygon* poly = new SUMO::Polygon(name, type, color, vec, fill, layer);
261  toFill.add(poly);
262  }
263  vec.clear();
264  }
265 }
