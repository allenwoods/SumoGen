6 // The Mersenne Twister is an algorithm for generating random numbers. It
7 // was designed with consideration of the flaws in various other generators.
8 // The period, 2^19937-1, and the order of equidistribution, 623 dimensions,
9 // are far greater. The generator is also fast; it avoids multiplication and
10 // division, and it benefits from caches and pipelines. For more information
11 // see the inventors' web page at http://www.math.keio.ac.jp/~matumoto/emt.html

180 inline double MTRand::randNorm( const double& mean, const double& variance )
181 {
182  // Return a real number from a normal (Gaussian) distribution with given
183  // mean and variance by Box-Muller method
184  double r = sqrt( -2.0 * log( 1.0-randDblExc()) ) * variance;
185  double phi = 2.0 * 3.14159265358979323846264338328 * randExc();
186  return mean + r * cos(phi);
187 }

189 inline MTRand::uint32 MTRand::randInt()
190 {
191  // Pull a 32-bit integer from the generator state
192  // Every other access function simply transforms the numbers extracted here
193 
194  if( left == 0 ) reload();
195  --left;
196 
197  register uint32 s1;
198  s1 = *pNext++;
199  s1 ^= (s1 >> 11);
200  s1 ^= (s1 << 7) & 0x9d2c5680UL;
201  s1 ^= (s1 << 15) & 0xefc60000UL;
202  return ( s1 ^ (s1 >> 18) );
203 }

205 inline MTRand::uint32 MTRand::randInt( const uint32& n )
206 {
207  // Find which bits are used in n
208  // Optimized by Magnus Jonsson (magnus@smartelectronix.com)
209  uint32 used = n;
210  used |= used >> 1;
211  used |= used >> 2;
212  used |= used >> 4;
213  used |= used >> 8;
214  used |= used >> 16;
215 
216  // Draw numbers until one is found in [0,n]
217  uint32 i;
218  do
219  i = randInt() & used; // toss unused bits to shorten search
220  while( i > n );
221  return i;
222 }

257 inline void MTRand::seed( uint32 *const bigSeed, const uint32 seedLength )
258 {
259  // Seed the generator with an array of uint32's
260  // There are 2^19937-1 possible initial states. This function allows
261  // all of those to be accessed by providing at least 19937 bits (with a
262  // default seed length of N = 624 uint32's). Any bits above the lower 32
263  // in each element are discarded.
264  // Just call seed() if you want to get array from /dev/urandom
265  initialize(19650218UL);
266  register int i = 1;
267  register uint32 j = 0;
268  register uint32 k = N;
269  if( seedLength > k ) k = seedLength;
270  for( ; k; --k )
271  {
272  state[i] =
273  state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1664525UL );
274  state[i] += ( bigSeed[j] & 0xffffffffUL ) + j;
275  state[i] &= 0xffffffffUL;
276  ++i; ++j;
277  if( i >= N ) { state[0] = state[N-1]; i = 1; }
278  if( j >= seedLength ) j = 0;
279  }
280  for( k = N - 1; k; --k )
281  {
282  state[i] =
283  state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1566083941UL );
284  state[i] -= i;
285  state[i] &= 0xffffffffUL;
286  ++i;
287  if( i >= N ) { state[0] = state[N-1]; i = 1; }
288  }
289  state[0] = 0x80000000UL; // MSB is 1, assuring non-zero initial array
290  reload();
291 }

294 inline void MTRand::seed()
295 {
296  // Seed the generator with an array from /dev/urandom if available
297  // Otherwise use a hash of time() and clock() values
298 
299  // First try getting an array from /dev/urandom
300  FILE* urandom = fopen( "/dev/urandom", "rb" );
301  if( urandom )
302  {
303  uint32 bigSeed[N];
304  register uint32 *s = bigSeed;
305  register int i = N;
306  register bool success = true;
307  while( success && i-- )
308  success = fread( s++, sizeof(uint32), 1, urandom )!=0; // !!! dk 16.02.2007
309  fclose(urandom);
310  if( success ) { seed( bigSeed, N ); return; }
311  }
312 
313  // Was not successful, so use time() and clock() instead
314  seed( hash( time(NULL), clock() ) );
315 }

318 inline void MTRand::initialize( const uint32 seed )
319 {
320  // Initialize generator state with seed
321  // See Knuth TAOCP Vol 2, 3rd Ed, p.106 for multiplier.
322  // In previous versions, most significant bits (MSBs) of the seed affect
323  // only MSBs of the state array. Modified 9 Jan 2002 by Makoto Matsumoto.
324  register uint32 *s = state;
325  register uint32 *r = state;
326  register int i = 1;
327  *s++ = seed & 0xffffffffUL;
328  for( ; i < N; ++i )
329  {
330  *s++ = ( 1812433253UL * ( *r ^ (*r >> 30) ) + i ) & 0xffffffffUL;
331  r++;
332  }
333 }

336 inline void MTRand::reload()
337 {
338  // Generate N new values in state
339  // Made clearer and faster by Matthew Bellew (matthew.bellew@home.com)
340  register uint32 *p = state;
341  register int i;
342  for( i = N - M; i--; ++p )
343  *p = twist( p[M], p[0], p[1] );
344  for( i = M; --i; ++p )
345  *p = twist( p[M-N], p[0], p[1] );
346  *p = twist( p[M-N], p[0], state[0] );
347 
348  left = N, pNext = state;
349 }

352 inline MTRand::uint32 MTRand::hash( time_t t, clock_t c )
353 {
354  // Get a uint32 from t and c
355  // Better than uint32(x) in case x is floating point in [0,1]
356  // Based on code by Lawrence Kirby (fred@genesis.demon.co.uk)
357 
358  static uint32 differ = 0; // guarantee time-based seeds will change
359 
360  uint32 h1 = 0;
361  unsigned char *p = (unsigned char *) &t;
362  for( size_t i = 0; i < sizeof(t); ++i )
363  {
364  h1 *= UCHAR_MAX + 2U;
365  h1 += p[i];
366  }
367  uint32 h2 = 0;
368  p = (unsigned char *) &c;
369  for( size_t j = 0; j < sizeof(c); ++j )
370  {
371  h2 *= UCHAR_MAX + 2U;
372  h2 += p[j];
373  }
374  return ( h1 + differ++ ) ^ h2;
375 }
