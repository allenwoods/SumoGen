60 Polygon::writeXML(OutputDevice& out, bool geo) {
61  out.openTag(SUMO_TAG_POLY);
62  out.writeAttr(SUMO_ATTR_ID, StringUtils::escapeXML(getID()));
63  out.writeAttr(SUMO_ATTR_TYPE, StringUtils::escapeXML(getType()));
64  out.writeAttr(SUMO_ATTR_COLOR, getColor());
65  out.writeAttr(SUMO_ATTR_FILL, getFill());
66  out.writeAttr(SUMO_ATTR_LAYER, getLayer());
67  PositionVector shape = getShape();
68  if (geo) {
69  out.writeAttr(SUMO_ATTR_GEO, true);
70  for (int i = 0; i < (int) shape.size(); i++) {
71  GeoConvHelper::getFinal().cartesian2geo(shape[i]);
72  }
73  }
74  out.writeAttr(SUMO_ATTR_SHAPE, shape);
75  if (getNaviDegree() != Shape::DEFAULT_ANGLE) {
76  out.writeAttr(SUMO_ATTR_ANGLE, getNaviDegree());
77  }
78  if (getImgFile() != Shape::DEFAULT_IMG_FILE) {
79  out.writeAttr(SUMO_ATTR_IMGFILE, getImgFile());
80  }
81  for (std::map<std::string, std::string>::const_iterator j = getMap().begin(); j != getMap().end(); ++j) {
82  out.openTag(SUMO_TAG_PARAM);
83  out.writeAttr(SUMO_ATTR_KEY, (*j).first);
84  out.writeAttr(SUMO_ATTR_VALUE, (*j).second);
85  out.closeTag();
86  }
87  out.closeTag();
88 }
