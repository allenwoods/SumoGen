67 NIXMLTypesHandler::myStartElement(int element,
68  const SUMOSAXAttributes& attrs) {
69  switch (element) {
70  case SUMO_TAG_TYPE: {
71  bool ok = true;
72  // get the id, report a warning if not given or empty...
73  myCurrentTypeID = attrs.get<std::string>(SUMO_ATTR_ID, 0, ok);
74  const char* const id = myCurrentTypeID.c_str();
75  const std::string defType = myTypeCont.knows(myCurrentTypeID) ? myCurrentTypeID : "";
76  const int priority = attrs.getOpt<int>(SUMO_ATTR_PRIORITY, id, ok, myTypeCont.getPriority(defType));
77  const int numLanes = attrs.getOpt<int>(SUMO_ATTR_NUMLANES, id, ok, myTypeCont.getNumLanes(defType));
78  const SUMOReal speed = attrs.getOpt<SUMOReal>(SUMO_ATTR_SPEED, id, ok, myTypeCont.getSpeed(defType));
79  const std::string allowS = attrs.getOpt<std::string>(SUMO_ATTR_ALLOW, id, ok, "");
80  const std::string disallowS = attrs.getOpt<std::string>(SUMO_ATTR_DISALLOW, id, ok, "");
81  const bool oneway = attrs.getOpt<bool>(SUMO_ATTR_ONEWAY, id, ok, myTypeCont.getIsOneWay(defType));
82  const bool discard = attrs.getOpt<bool>(SUMO_ATTR_DISCARD, id, ok, false);
83  const SUMOReal width = attrs.getOpt<SUMOReal>(SUMO_ATTR_WIDTH, id, ok, myTypeCont.getWidth(defType));
84  const SUMOReal sidewalkWidth = attrs.getOpt<SUMOReal>(SUMO_ATTR_SIDEWALKWIDTH, id, ok, myTypeCont.getSidewalkWidth(defType));
85  const SUMOReal bikeLaneWidth = attrs.getOpt<SUMOReal>(SUMO_ATTR_BIKELANEWIDTH, id, ok, myTypeCont.getBikeLaneWidth(defType));
86  if (!ok) {
87  return;
88  }
89  // build the type
90  SVCPermissions permissions = myTypeCont.getPermissions(defType);
91  if (allowS != "" || disallowS != "") {
92  permissions = parseVehicleClasses(allowS, disallowS);
93  }
94  myTypeCont.insert(myCurrentTypeID, numLanes, speed, priority, permissions, width, oneway, sidewalkWidth, bikeLaneWidth);
95  if (discard) {
96  myTypeCont.markAsToDiscard(myCurrentTypeID);
97  }
98  SumoXMLAttr myAttrs[] = {SUMO_ATTR_PRIORITY, SUMO_ATTR_NUMLANES, SUMO_ATTR_SPEED,
99  SUMO_ATTR_ALLOW, SUMO_ATTR_DISALLOW, SUMO_ATTR_ONEWAY,
100  SUMO_ATTR_DISCARD, SUMO_ATTR_WIDTH, SUMO_ATTR_SIDEWALKWIDTH, SUMO_ATTR_BIKELANEWIDTH
101  };
102  for (int i = 0; i < 10; i++) {
103  if (attrs.hasAttribute(myAttrs[i])) {
104  myTypeCont.markAsSet(myCurrentTypeID, myAttrs[i]);
105  }
106  }
107  break;
108  }
109  case SUMO_TAG_RESTRICTION: {
110  bool ok = true;
111  const SUMOVehicleClass svc = getVehicleClassID(attrs.get<std::string>(SUMO_ATTR_VCLASS, myCurrentTypeID.c_str(), ok));
112  const SUMOReal speed = attrs.get<SUMOReal>(SUMO_ATTR_SPEED, myCurrentTypeID.c_str(), ok);
113  if (ok) {
114  myTypeCont.addRestriction(myCurrentTypeID, svc, speed);
115  }
116  break;
117  }
118  default:
119  break;
120  }
121 }
