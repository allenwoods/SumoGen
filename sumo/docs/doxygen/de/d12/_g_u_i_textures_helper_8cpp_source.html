124 int
125 GUITexturesHelper::getTextureID(const std::string& filename, const bool mirrorX) {
126  if (myTextures.count(filename) == 0) {
127  try {
128  FXImage* i = MFXImageHelper::loadImage(GUIMainWindow::getInstance()->getApp(), filename);
129  if (mirrorX) {
130  i->mirror(false, true);
131  }
132  MFXImageHelper::scalePower2(i, getMaxTextureSize());
133  GUIGlID id = add(i);
134  delete i;
135  myTextures[filename] = (int)id;
136  } catch (InvalidArgument& e) {
137  WRITE_ERROR("Could not load '" + filename + "'.\n" + e.what());
138  myTextures[filename] = -1;
139  }
140  }
141  return myTextures[filename];
142 }
