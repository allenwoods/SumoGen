151 NBContHelper::relative_outgoing_edge_sorter::operator()(NBEdge* e1, NBEdge* e2) const {
152  if (e1 == 0 || e2 == 0) {
153  return -1;
154  }
155  SUMOReal relAngle1 = NBHelpers::normRelAngle(
156  myEdge->getEndAngle(), e1->getStartAngle());
157  SUMOReal relAngle2 = NBHelpers::normRelAngle(
158  myEdge->getEndAngle(), e2->getStartAngle());
159 
160  SUMOReal lookAhead = 2 * NBEdge::ANGLE_LOOKAHEAD;
161  while (fabs(relAngle1 - relAngle2) < 3.0) {
162  // look at further geometry segments to resolve ambiguity
163  const Position referencePos1 = e1->getGeometry().positionAtOffset2D(lookAhead);
164  const Position referencePos2 = e2->getGeometry().positionAtOffset2D(lookAhead);
165  relAngle1 = NBHelpers::normRelAngle(myEdge->getEndAngle(), GeomHelper::legacyDegree(
166  e1->getFromNode()->getPosition().angleTo2D(referencePos1), true));
167  relAngle2 = NBHelpers::normRelAngle(myEdge->getEndAngle(), GeomHelper::legacyDegree(
168  e2->getFromNode()->getPosition().angleTo2D(referencePos2), true));
169  if (lookAhead > MAX2(e1->getLength(), e2->getLength())) {
170  break;
171  }
172  lookAhead *= 2;
173  }
174  return relAngle1 > relAngle2;
175 }

296 int
297 NBContHelper::edge_by_angle_to_nodeShapeCentroid_sorter::operator()(const NBEdge* e1, const NBEdge* e2) const {
298  assert(e1->getFromNode() == myNode || e1->getToNode() == myNode);
299  assert(e2->getFromNode() == myNode || e2->getToNode() == myNode);
300  const SUMOReal angle1 = e1->getAngleAtNodeToCenter(myNode);
301  const SUMOReal angle2 = e2->getAngleAtNodeToCenter(myNode);
302  const SUMOReal absDiff = fabs(angle1 - angle2);
303 
304  // cannot trust the angle difference hence a heuristic:
305  if (absDiff < 2 || absDiff > (360 - 2)) {
306  const bool sameDir = ((e1->getFromNode() == myNode && e2->getFromNode() == myNode)
307  || (e1->getToNode() == myNode && e2->getToNode() == myNode));
308  if (sameDir) {
309  // put edges that allow pedestrians on the 'outside', but be aware if both allow / disallow
310  const bool e1Peds = (e1->getPermissions() & SVC_PEDESTRIAN) != 0;
311  const bool e2Peds = (e2->getPermissions() & SVC_PEDESTRIAN) != 0;
312  if (e1->getToNode() == myNode) {
313  if (e1Peds && !e2Peds) {
314  return true;
315  } else if (!e1Peds && e2Peds) {
316  return false;
317  }
318  } else {
319  if (!e1Peds && e2Peds) {
320  return true;
321  } else if (e1Peds && !e2Peds) {
322  return false;
323  }
324  }
325  // break ties to ensure strictly weak ordering
326  return e1->getID() < e2->getID();
327  } else {
328  // sort incoming before outgoing, no need to break ties here
329  return e1->getToNode() == myNode;
330  }
331  }
332  return angle1 < angle2;
333 }
