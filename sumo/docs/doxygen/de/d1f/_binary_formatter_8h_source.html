263 template <typename E>
264 bool BinaryFormatter::writeHeader(std::ostream& into, const SumoXMLTag& rootElement) {
265  if (myXMLStack.empty()) {
266  writeStaticHeader(into);
267  const int numEdges = (const int)E::getAllEdges().size();
268  FileHelpers::writeByte(into, BF_LIST);
269  FileHelpers::writeInt(into, numEdges);
270  for (int i = 0; i < numEdges; i++) {
271  FileHelpers::writeByte(into, BF_STRING);
272  FileHelpers::writeString(into, E::getAllEdges()[i]->getID());
273  }
274  FileHelpers::writeByte(into, BF_LIST);
275  FileHelpers::writeInt(into, numEdges);
276  for (int i = 0; i < numEdges; i++) {
277  E* e = E::getAllEdges()[i];
278  FileHelpers::writeByte(into, BF_LIST);
279  FileHelpers::writeInt(into, e->getNumSuccessors());
280  for (int j = 0; j < e->getNumSuccessors(); j++) {
281  FileHelpers::writeByte(into, BF_INTEGER);
282  FileHelpers::writeInt(into, e->getSuccessors()[j]->getNumericalID());
283  }
284  }
285  openTag(into, rootElement);
286  return true;
287  }
288  return false;
289 }
