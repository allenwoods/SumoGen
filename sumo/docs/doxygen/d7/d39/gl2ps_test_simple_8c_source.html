94 static void keyboard(unsigned char key, int x, int y)
95 {
96  FILE *fp;
97  int state = GL2PS_OVERFLOW, buffsize = 0;
98 
99  (void) x; (void) y; /* not used */
100  switch(key){
101  case 'q':
102  exit(0);
103  break;
104  case 's':
105  fp = fopen("out.eps", "wb");
106  printf("Writing 'out.eps'... ");
107  while(state == GL2PS_OVERFLOW){
108  buffsize += 1024*1024;
109  gl2psBeginPage("test", "gl2psTestSimple", NULL, GL2PS_EPS, GL2PS_SIMPLE_SORT,
110  GL2PS_DRAW_BACKGROUND | GL2PS_USE_CURRENT_VIEWPORT,
111  GL_RGBA, 0, NULL, 0, 0, 0, buffsize, fp, "out.eps");
112  display();
113  state = gl2psEndPage();
114  }
115  fclose(fp);
116  printf("Done!\n");
117  break;
118  }
