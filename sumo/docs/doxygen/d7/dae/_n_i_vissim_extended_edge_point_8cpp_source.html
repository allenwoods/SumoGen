82  // check whether an "all" indicator is there
83  bool hadAll = false;
84  for (std::vector<int>::const_iterator i = myLanes.begin(); !hadAll && i != myLanes.end(); ++i) {

86  hadAll = true;
87  }
88  }
89  // no -> return

91  return;
92  }
93  // patch lane indices
94  myLanes.clear();
