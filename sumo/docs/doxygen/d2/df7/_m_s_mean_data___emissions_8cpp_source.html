92 MSMeanData_Emissions::MSLaneMeanDataValues::write(OutputDevice& dev, const SUMOTime period,
93  const SUMOReal /*numLanes*/, const SUMOReal defaultTravelTime, const int /*numVehicles*/) const {
94  const SUMOReal normFactor = SUMOReal(3600. / STEPS2TIME(period) / myLaneLength);
95  dev << " CO_abs=\"" << OutputDevice::realString(myEmissions.CO, 6) <<
96  "\" CO2_abs=\"" << OutputDevice::realString(myEmissions.CO2, 6) <<
97  "\" HC_abs=\"" << OutputDevice::realString(myEmissions.HC, 6) <<
98  "\" PMx_abs=\"" << OutputDevice::realString(myEmissions.PMx, 6) <<
99  "\" NOx_abs=\"" << OutputDevice::realString(myEmissions.NOx, 6) <<
100  "\" fuel_abs=\"" << OutputDevice::realString(myEmissions.fuel, 6) <<
101  "\" electricity_abs=\"" << OutputDevice::realString(myEmissions.electricity, 6) <<
102  "\"\n CO_normed=\"" << OutputDevice::realString(normFactor * myEmissions.CO, 6) <<
103  "\" CO2_normed=\"" << OutputDevice::realString(normFactor * myEmissions.CO2, 6) <<
104  "\" HC_normed=\"" << OutputDevice::realString(normFactor * myEmissions.HC, 6) <<
105  "\" PMx_normed=\"" << OutputDevice::realString(normFactor * myEmissions.PMx, 6) <<
106  "\" NOx_normed=\"" << OutputDevice::realString(normFactor * myEmissions.NOx, 6) <<
107  "\" fuel_normed=\"" << OutputDevice::realString(normFactor * myEmissions.fuel, 6) <<
108  "\" electricity_normed=\"" << OutputDevice::realString(normFactor * myEmissions.electricity, 6);
109  if (sampleSeconds > myParent->myMinSamples) {
110  SUMOReal vehFactor = myParent->myMaxTravelTime / sampleSeconds;
111  SUMOReal traveltime = myParent->myMaxTravelTime;
112  if (travelledDistance > 0.f) {
113  vehFactor = MIN2(vehFactor, myLaneLength / travelledDistance);
114  traveltime = MIN2(traveltime, myLaneLength * sampleSeconds / travelledDistance);
115  }
116  dev << "\"\n traveltime=\"" << OutputDevice::realString(traveltime) <<
117  "\" CO_perVeh=\"" << OutputDevice::realString(myEmissions.CO * vehFactor, 6) <<
118  "\" CO2_perVeh=\"" << OutputDevice::realString(myEmissions.CO2 * vehFactor, 6) <<
119  "\" HC_perVeh=\"" << OutputDevice::realString(myEmissions.HC * vehFactor, 6) <<
120  "\" PMx_perVeh=\"" << OutputDevice::realString(myEmissions.PMx * vehFactor, 6) <<
121  "\" NOx_perVeh=\"" << OutputDevice::realString(myEmissions.NOx * vehFactor, 6) <<
122  "\" fuel_perVeh=\"" << OutputDevice::realString(myEmissions.fuel * vehFactor, 6) <<
123  "\" electricity_perVeh=\"" << OutputDevice::realString(myEmissions.electricity * vehFactor, 6);
124  } else if (defaultTravelTime >= 0.) {
125  const MSVehicleType* t = MSNet::getInstance()->getVehicleControl().getVType();
126  const SUMOReal speed = MIN2(myLaneLength / defaultTravelTime, t->getMaxSpeed());
127  dev << "\"\n traveltime=\"" << OutputDevice::realString(defaultTravelTime) <<
128  "\" CO_perVeh=\"" << OutputDevice::realString(PollutantsInterface::computeDefault(t->getEmissionClass(), PollutantsInterface::CO, speed, t->getCarFollowModel().getMaxAccel(), 0, defaultTravelTime), 6) << // @todo: give correct slope
129  "\" CO2_perVeh=\"" << OutputDevice::realString(PollutantsInterface::computeDefault(t->getEmissionClass(), PollutantsInterface::CO2, speed, t->getCarFollowModel().getMaxAccel(), 0, defaultTravelTime), 6) << // @todo: give correct slope
130  "\" HC_perVeh=\"" << OutputDevice::realString(PollutantsInterface::computeDefault(t->getEmissionClass(), PollutantsInterface::HC, speed, t->getCarFollowModel().getMaxAccel(), 0, defaultTravelTime), 6) << // @todo: give correct slope
131  "\" PMx_perVeh=\"" << OutputDevice::realString(PollutantsInterface::computeDefault(t->getEmissionClass(), PollutantsInterface::PM_X, speed, t->getCarFollowModel().getMaxAccel(), 0, defaultTravelTime), 6) << // @todo: give correct slope
132  "\" NOx_perVeh=\"" << OutputDevice::realString(PollutantsInterface::computeDefault(t->getEmissionClass(), PollutantsInterface::NO_X, speed, t->getCarFollowModel().getMaxAccel(), 0, defaultTravelTime), 6) << // @todo: give correct slope
133  "\" fuel_perVeh=\"" << OutputDevice::realString(PollutantsInterface::computeDefault(t->getEmissionClass(), PollutantsInterface::FUEL, speed, t->getCarFollowModel().getMaxAccel(), 0, defaultTravelTime), 6) << // @todo: give correct slope
134  "\" electricity_perVeh=\"" << OutputDevice::realString(PollutantsInterface::computeDefault(t->getEmissionClass(), PollutantsInterface::ELEC, speed, t->getCarFollowModel().getMaxAccel(), 0, defaultTravelTime), 6); // @todo: give correct slope
135  }
136  dev << "\"";
137  dev.closeTag();
138 }
