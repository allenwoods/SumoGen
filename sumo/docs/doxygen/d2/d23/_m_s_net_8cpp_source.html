306 MSNet::simulate(SUMOTime start, SUMOTime stop) {
307  // report the begin when wished
308  WRITE_MESSAGE("Simulation started with time: " + time2string(start));
309  // the simulation loop
310  MSNet::SimulationState state = SIMSTATE_RUNNING;
311  myStep = start;
312  // preload the routes especially for TraCI
313  loadRoutes();
314 #ifndef NO_TRACI
315 #ifdef HAVE_PYTHON
316  if (OptionsCont::getOptions().isSet("python-script")) {
317  TraCIServer::runEmbedded(OptionsCont::getOptions().getString("python-script"));
318  closeSimulation(start);
319  WRITE_MESSAGE("Simulation ended at time: " + time2string(getCurrentTimeStep()));
320  WRITE_MESSAGE("Reason: Script ended");
321  return 0;
322  }
323 #endif
324 #endif
325  while (state == SIMSTATE_RUNNING) {
326  if (myLogStepNumber) {
327  preSimStepOutput();
328  }
329  simulationStep();
330  if (myLogStepNumber) {
331  postSimStepOutput();
332  }
333  state = simulationState(stop);
334 #ifndef NO_TRACI
335  if (state != SIMSTATE_RUNNING) {
336  if (OptionsCont::getOptions().getInt("remote-port") != 0 && !TraCIServer::wasClosed()) {
337  state = SIMSTATE_RUNNING;
338  }
339  }
340 #endif
341  }
342  // report the end when wished
343  WRITE_MESSAGE("Simulation ended at time: " + time2string(getCurrentTimeStep()));
344  WRITE_MESSAGE("Reason: " + getStateMessage(state));
345  // exit simulation loop
346  closeSimulation(start);
347  return 0;
348 }

423 void
424 MSNet::simulationStep() {
425 #ifndef NO_TRACI
426  TraCIServer::processCommandsUntilSimStep(myStep);
427  TraCIServer* t = TraCIServer::getInstance();
428  if (t != 0 && t->getTargetTime() != 0 && t->getTargetTime() < myStep) {
429  return;
430  }
431 #endif
432  // execute beginOfTimestepEvents
433  if (myLogExecutionTime) {
434  mySimStepBegin = SysUtils::getCurrentMillis();
435  }
436  // simulation state output
437  std::vector<SUMOTime>::iterator timeIt = find(myStateDumpTimes.begin(), myStateDumpTimes.end(), myStep);
438  if (timeIt != myStateDumpTimes.end()) {
439  const int dist = (int)distance(myStateDumpTimes.begin(), timeIt);
440  MSStateHandler::saveState(myStateDumpFiles[dist], myStep);
441  }
442  if (myStateDumpPeriod > 0 && myStep % myStateDumpPeriod == 0) {
443  MSStateHandler::saveState(myStateDumpPrefix + "_" + time2string(myStep) + myStateDumpSuffix, myStep);
444  }
445  myBeginOfTimestepEvents->execute(myStep);
446 #ifdef HAVE_FOX
447  MSDevice_Routing::waitForAll();
448 #endif
449  if (MSGlobals::gCheck4Accidents) {
450  myEdges->detectCollisions(myStep, STAGE_EVENTS);
451  }
452  // check whether the tls programs need to be switched
453  myLogics->check2Switch(myStep);
454 
455  if (MSGlobals::gUseMesoSim) {
456  MSGlobals::gMesoNet->simulate(myStep);
457  } else {
458  // assure all lanes with vehicles are 'active'
459  myEdges->patchActiveLanes();
460 
461  // compute safe velocities for all vehicles for the next few lanes
462  // also register ApproachingVehicleInformation for all links
463  myEdges->planMovements(myStep);
464 
465  // decide right-of-way and execute movements
466  myEdges->executeMovements(myStep);
467  if (MSGlobals::gCheck4Accidents) {
468  myEdges->detectCollisions(myStep, STAGE_MOVEMENTS);
469  }
470 
471  // vehicles may change lanes
472  myEdges->changeLanes(myStep);
473 
474  if (MSGlobals::gCheck4Accidents) {
475  myEdges->detectCollisions(myStep, STAGE_LANECHANGE);
476  }
477  }
478  loadRoutes();
479 
480  // persons
481  if (myPersonControl != 0 && myPersonControl->hasTransportables()) {
482  myPersonControl->checkWaiting(this, myStep);
483  }
484  // containers
485  if (myContainerControl != 0 && myContainerControl->hasTransportables()) {
486  myContainerControl->checkWaiting(this, myStep);
487  }
488  // insert vehicles
489  myInserter->determineCandidates(myStep);
490  myInsertionEvents->execute(myStep);
491 #ifdef HAVE_FOX
492  MSDevice_Routing::waitForAll();
493 #endif
494  myInserter->emitVehicles(myStep);
495  if (MSGlobals::gCheck4Accidents) {
496  //myEdges->patchActiveLanes(); // @note required to detect collisions on lanes that were empty before insertion. wasteful?
497  myEdges->detectCollisions(myStep, STAGE_INSERTIONS);
498  }
499  MSVehicleTransfer::getInstance()->checkInsertions(myStep);
500 
501  // execute endOfTimestepEvents
502  myEndOfTimestepEvents->execute(myStep);
503 
504 #ifndef NO_TRACI
505  if (TraCIServer::getInstance() != 0) {
506  TraCIServer::getInstance()->postProcessVTD();
507  }
508 #endif
509  // update and write (if needed) detector values
510  writeOutput();
511 
512  if (myLogExecutionTime) {
513  mySimStepEnd = SysUtils::getCurrentMillis();
514  mySimStepDuration = mySimStepEnd - mySimStepBegin;
515  myVehiclesMoved += myVehicleControl->getRunningVehicleNo();
516  }
517  myStep += DELTA_T;
518 }

521 MSNet::SimulationState
522 MSNet::simulationState(SUMOTime stopTime) const {
523 #ifndef NO_TRACI
524  if (TraCIServer::wasClosed()) {
525  return SIMSTATE_CONNECTION_CLOSED;
526  }
527  if (stopTime < 0 && OptionsCont::getOptions().getInt("remote-port") == 0) {
528 #else
529  if (stopTime < 0) {
530 #endif
531  if (myInsertionEvents->isEmpty()
532  && (myVehicleControl->getActiveVehicleCount() == 0)
533  && (myInserter->getPendingFlowCount() == 0)
534  && (myPersonControl == 0 || !myPersonControl->hasNonWaiting())
535  && (myContainerControl == 0 || !myContainerControl->hasNonWaiting())) {
536  if (myPersonControl) {
537  myPersonControl->abortWaiting();
538  }
539  if (myContainerControl) {
540  myContainerControl->abortWaiting();
541  }
542  myVehicleControl->abortWaiting();
543  return SIMSTATE_NO_FURTHER_VEHICLES;
544  }
545  }
546  if (stopTime >= 0 && myStep >= stopTime) {
547  return SIMSTATE_END_STEP_REACHED;
548  }
549  return SIMSTATE_RUNNING;
550 }
