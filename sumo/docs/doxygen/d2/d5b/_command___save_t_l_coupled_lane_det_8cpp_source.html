64 void
65 Command_SaveTLCoupledLaneDet::execute() {
66  if (myLink->getState() == myLastState && myHadOne) {
67  return;
68  }
69  myHadOne = true;
70  if (myLastState == LINKSTATE_TL_RED && myLink->getState() != LINKSTATE_TL_RED) {
71  SUMOTime end = MSNet::getInstance()->getCurrentTimeStep();
72  if (myStartTime != end) {
73  myDetector->writeXMLOutput(myDevice, myStartTime, end);
74  myStartTime = end;
75  }
76  } else if (myLink->getState() == LINKSTATE_TL_RED) {
77  myDetector->reset();
78  myStartTime = MSNet::getInstance()->getCurrentTimeStep();
79  }
80  myLastState = myLink->getState();
81 }
