51 ODAmitranHandler::myStartElement(int element, const SUMOSAXAttributes& attrs) {
52  bool ok = true;
53  switch (element) {
54  case SUMO_TAG_ACTORCONFIG:
55  myVehicleType = attrs.get<std::string>(SUMO_ATTR_ID, 0, ok);
56  break;
57  case SUMO_TAG_TIMESLICE:
58  myBegin = attrs.get<int>(SUMO_ATTR_STARTTIME, myVehicleType.c_str(), ok);
59  myEnd = myBegin + attrs.get<int>(SUMO_ATTR_DURATION, myVehicleType.c_str(), ok);
60  if (myBegin >= myEnd) {
61  WRITE_ERROR("Invalid duration for timeSlice starting " + toString(myBegin) + ".");
62  }
63  break;
64  case SUMO_TAG_OD_PAIR:
65  myMatrix.add(attrs.get<SUMOReal>(SUMO_ATTR_AMOUNT, myVehicleType.c_str(), ok),
66  myBegin, myEnd, attrs.get<std::string>(SUMO_ATTR_ORIGIN, myVehicleType.c_str(), ok),
67  attrs.get<std::string>(SUMO_ATTR_DESTINATION, myVehicleType.c_str(), ok), myVehicleType);
68  break;
69  default:
70  break;
71  }
