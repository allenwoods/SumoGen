54 void
55 NWWriter_Amitran::writeNetwork(const OptionsCont& oc, NBNetBuilder& nb) {
56  // check whether an amitran-file shall be generated
57  if (!oc.isSet("amitran-output")) {
58  return;
59  }
60  NBEdgeCont& ec = nb.getEdgeCont();
61  OutputDevice& device = OutputDevice::getDevice(oc.getString("amitran-output"));
62  device << "<?xml version=\"1.0\" encoding=\"utf-8\"?>\n";
63  device << "<network xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:noNamespaceSchemaLocation=\"http://sumo.dlr.de/xsd/amitran/network.xsd\">\n";
64  // write nodes
65  int index = 0;
66  NBNodeCont& nc = nb.getNodeCont();
67  std::set<NBNode*> singleRoundaboutNodes;
68  std::set<NBNode*> multiRoundaboutNodes;
69  const std::set<EdgeSet>& roundabouts = ec.getRoundabouts();
70  for (std::set<EdgeSet>::const_iterator i = roundabouts.begin(); i != roundabouts.end(); ++i) {
71  for (EdgeSet::const_iterator j = (*i).begin(); j != (*i).end(); ++j) {
72  if ((*j)->getNumLanes() > 1) {
73  multiRoundaboutNodes.insert((*j)->getFromNode());
74  } else {
75  singleRoundaboutNodes.insert((*j)->getFromNode());
76  }
77  }
78  }
79  std::map<NBNode*, int> nodeIds;
80  for (std::map<std::string, NBNode*>::const_iterator i = nc.begin(); i != nc.end(); ++i) {
81  device << " <node id=\"" << index;
82  nodeIds[i->second] = index++;
83  if (singleRoundaboutNodes.count(i->second) > 0) {
84  device << "\" type=\"roundaboutSingle\"/>\n";
85  continue;
86  }
87  if (multiRoundaboutNodes.count(i->second) > 0) {
88  device << "\" type=\"roundaboutMulti\"/>\n";
89  continue;
90  }
91  switch (i->second->getType()) {
92  case NODETYPE_TRAFFIC_LIGHT:
93  case NODETYPE_TRAFFIC_LIGHT_NOJUNCTION:
94  case NODETYPE_TRAFFIC_LIGHT_RIGHT_ON_RED:
95  device << "\" type=\"trafficLight";
96  break;
97  case NODETYPE_PRIORITY:
98  device << "\" type=\"priority";
99  break;
100  case NODETYPE_PRIORITY_STOP:
101  device << "\" type=\"priorityStop";
102  break;
103  case NODETYPE_RIGHT_BEFORE_LEFT:
104  device << "\" type=\"rightBeforeLeft";
105  break;
106  case NODETYPE_ALLWAY_STOP:
107  device << "\" type=\"allwayStop";
108  break;
109  case NODETYPE_ZIPPER:
110  device << "\" type=\"zipper";
111  break;
112  case NODETYPE_RAIL_SIGNAL:
113  device << "\" type=\"railSignal";
114  break;
115  case NODETYPE_RAIL_CROSSING:
116  device << "\" type=\"railCrossing";
117  break;
118  case NODETYPE_DEAD_END:
119  case NODETYPE_DEAD_END_DEPRECATED:
120  device << "\" type=\"deadEnd";
121  break;
122  case NODETYPE_DISTRICT:
123  case NODETYPE_NOJUNCTION:
124  case NODETYPE_INTERNAL:
125  case NODETYPE_UNKNOWN:
126  break;
127  }
128  device << "\"/>\n";
129  }
130  // write edges
131  index = 0;
132  for (std::map<std::string, NBEdge*>::const_iterator i = ec.begin(); i != ec.end(); ++i) {
133  device << " <link id=\"" << index++
134  << "\" from=\"" << nodeIds[i->second->getFromNode()]
135  << "\" to=\"" << nodeIds[i->second->getToNode()]
136  << "\" roadClass=\"" << NWWriter_DlrNavteq::getRoadClass((*i).second)
137  << "\" length=\"" << int(1000 * i->second->getLoadedLength())
138  << "\" speedLimitKmh=\"" << int(3.6 * (*i).second->getSpeed() + 0.5)
139  << "\" laneNr=\"" << (*i).second->getNumLanes()
140  << "\"/>\n";
141  }
142  device << "</network>\n";
143  device.close();
144 }
