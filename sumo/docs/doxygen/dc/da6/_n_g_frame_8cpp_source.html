193 bool
194 NGFrame::checkOptions() {
195  OptionsCont& oc = OptionsCont::getOptions();
196  bool ok = true;
197  // check whether exactly one type of a network to build was wished
198  int no = 0;
199  if (oc.getBool("spider")) {
200  no++;
201  }
202  if (oc.getBool("grid")) {
203  no++;
204  }
205  if (oc.getBool("rand")) {
206  no++;
207  }
208  if (no == 0) {
209  WRITE_ERROR("You have to specify the type of network to generate.");
210  ok = false;
211  }
212  if (no > 1) {
213  WRITE_ERROR("You may specify only one type of network to generate at once.");
214  ok = false;
215  }
216  // check whether the junction type to use is properly set
217  if (oc.isSet("default-junction-type")) {
218  std::string type = oc.getString("default-junction-type");
219  if (type != toString(NODETYPE_TRAFFIC_LIGHT) &&
220  type != toString(NODETYPE_TRAFFIC_LIGHT_NOJUNCTION) &&
221  type != toString(NODETYPE_TRAFFIC_LIGHT_RIGHT_ON_RED) &&
222  type != toString(NODETYPE_PRIORITY) &&
223  type != toString(NODETYPE_PRIORITY_STOP) &&
224  type != toString(NODETYPE_ALLWAY_STOP) &&
225  type != toString(NODETYPE_ZIPPER) &&
226  type != toString(NODETYPE_NOJUNCTION) &&
227  type != toString(NODETYPE_RAIL_SIGNAL) &&
228  type != toString(NODETYPE_RAIL_CROSSING) &&
229  type != toString(NODETYPE_RIGHT_BEFORE_LEFT)) {
230  WRITE_ERROR("Only the following junction types are known: " +
231  toString(NODETYPE_TRAFFIC_LIGHT) + ", " +
232  toString(NODETYPE_TRAFFIC_LIGHT_NOJUNCTION) + ", " +
233  toString(NODETYPE_TRAFFIC_LIGHT_RIGHT_ON_RED) + ", " +
234  toString(NODETYPE_PRIORITY) + ", " +
235  toString(NODETYPE_PRIORITY_STOP) + ", " +
236  toString(NODETYPE_ALLWAY_STOP) + ", " +
237  toString(NODETYPE_ZIPPER) + ", " +
238  toString(NODETYPE_NOJUNCTION) + ", " +
239  toString(NODETYPE_RAIL_SIGNAL) + ", " +
240  toString(NODETYPE_RAIL_CROSSING) + ", " +
241  toString(NODETYPE_RIGHT_BEFORE_LEFT));
242  ok = false;
243  }
244  }
245  return ok;
246 }
