90 GUIRunThread::init(GUINet* net, SUMOTime start, SUMOTime end) {
91  assert(net != 0);
92  // assign new values
93  myOk = true;
94  myNet = net;
95  mySimStartTime = start;
96  mySimEndTime = end;
97  // register message callbacks

100  if (!OptionsCont::getOptions().getBool("no-warnings")) {
101  MsgHandler::getWarningInstance()->addRetriever(myWarningRetriever);
102  }
103  // preload the routes especially for TraCI
104  mySimulationLock.lock();
105  try {
106  net->setCurrentTimeStep(start);
107  net->loadRoutes();
108  } catch (ProcessError& e2) {
109  if (std::string(e2.what()) != std::string("Process Error") && std::string(e2.what()) != std::string("")) {
110  WRITE_ERROR(e2.what());
111  }
112  MsgHandler::getErrorInstance()->inform("Quitting (on error).", false);
113  myHalting = true;
114  myOk = false;
115  mySimulationInProgress = false;
116 #ifndef _DEBUG
117  } catch (...) {
118  myHalting = true;
119  myOk = false;
120  mySimulationInProgress = false;
121 #endif
122  }
123  mySimulationLock.unlock();
124  return myOk;

128 FXint
129 GUIRunThread::run() {
130  long beg = 0;
131  long end = -1;
132  // perform an endless loop
133  while (!myQuit) {
134  // if the simulation shall be perfomed, do it
135  if (!myHalting && myNet != 0 && myOk) {
136  if (getNet().logSimulationDuration()) {
137  beg = SysUtils::getCurrentMillis();
138  if (end != -1) {
139  getNet().setIdleDuration((int)(beg - end));
140  }
141  }
142  // check whether we shall stop at this step
143  myBreakpointLock.lock();
144  const bool haltAfter = find(myBreakpoints.begin(), myBreakpoints.end(), myNet->getCurrentTimeStep()) != myBreakpoints.end();
145  myBreakpointLock.unlock();
146  // do the step
147  makeStep();
148  // stop if wished
149  if (haltAfter) {
150  stop();
151  }
152  // wait if wanted
153  long wait = (long) mySimDelay.getValue();
154  if (getNet().logSimulationDuration()) {
155  end = SysUtils::getCurrentMillis();
156  getNet().setSimDuration((int)(end - beg));
157  wait -= (end - beg);
158  }
159  if (wait > 0) {
160  sleep(wait);
161  }
162  } else {
163  // sleep if the simulation is not running
164  sleep(50);
165  }
166  }
167  // delete a maybe existing simulation at the end
168  deleteSim();
169  return 0;
170 }

173 void
174 GUIRunThread::makeStep() {
175  GUIEvent* e = 0;
176  // simulation is being perfomed
177  mySimulationInProgress = true;
178  // execute a single step
179  try {
180  mySimulationLock.lock();
181  myNet->simulationStep();
182  myNet->guiSimulationStep();
183  mySimulationLock.unlock();
184 
185  // inform parent that a step has been performed
186  e = new GUIEvent_SimulationStep();
187  myEventQue.add(e);
188  myEventThrow.signal();
189 
190  e = 0;
191  MSNet::SimulationState state = myNet->simulationState(mySimEndTime);
192 #ifndef NO_TRACI
193  if (state != MSNet::SIMSTATE_RUNNING) {
194  if (OptionsCont::getOptions().getInt("remote-port") != 0 && !TraCIServer::wasClosed()) {
195  state = MSNet::SIMSTATE_RUNNING;
196  }
197  }
198 #endif
199  switch (state) {
200  case MSNet::SIMSTATE_END_STEP_REACHED:
201  case MSNet::SIMSTATE_NO_FURTHER_VEHICLES:
202  case MSNet::SIMSTATE_CONNECTION_CLOSED:
203  case MSNet::SIMSTATE_TOO_MANY_VEHICLES:
204  WRITE_MESSAGE("Simulation ended at time: " + time2string(myNet->getCurrentTimeStep()));
205  WRITE_MESSAGE("Reason: " + MSNet::getStateMessage(state));
206  e = new GUIEvent_SimulationEnded(state, myNet->getCurrentTimeStep() - DELTA_T);
207  break;
208  default:
209  break;
210  }
211  if (e != 0) {
212  myEventQue.add(e);
213  myEventThrow.signal();
214  myHalting = true;
215  }
216  // stop the execution when only a single step should have
217  // been performed
218  if (mySingle) {
219  myHalting = true;
220  }
221  // simulation step is over
222  mySimulationInProgress = false;
223  } catch (ProcessError& e2) {
224  if (std::string(e2.what()) != std::string("Process Error") && std::string(e2.what()) != std::string("")) {
225  WRITE_ERROR(e2.what());
226  }
227  MsgHandler::getErrorInstance()->inform("Quitting (on error).", false);
228  mySimulationLock.unlock();
229  mySimulationInProgress = false;
230  e = new GUIEvent_SimulationEnded(MSNet::SIMSTATE_ERROR_IN_SIM, myNet->getCurrentTimeStep());
231  myEventQue.add(e);
232  myEventThrow.signal();
233  myHalting = true;
234  myOk = false;
235 #ifndef _DEBUG
236  } catch (...) {
237  mySimulationLock.unlock();
238  mySimulationInProgress = false;
239  e = new GUIEvent_SimulationEnded(MSNet::SIMSTATE_ERROR_IN_SIM, myNet->getCurrentTimeStep());
240  myEventQue.add(e);
241  myEventThrow.signal();
242  myHalting = true;
243  myOk = false;
244 #endif
245  }
246 }

285 GUIRunThread::deleteSim() {
286  myHalting = true;
287  // remove message callbacks
288  MsgHandler::getErrorInstance()->removeRetriever(myErrorRetriever);
289  MsgHandler::getWarningInstance()->removeRetriever(myWarningRetriever);
290  MsgHandler::getMessageInstance()->removeRetriever(myMessageRetriever);
291  //
292  mySimulationLock.lock();
293  if (myNet != 0) {
294  myNet->closeSimulation(mySimStartTime);
295  }
296  while (mySimulationInProgress);
297  delete myNet;
298  GUIGlObjectStorage::gIDStorage.clear();
299  myNet = 0;

301  mySimulationLock.unlock();
302  MsgHandler::cleanupOnEnd();
