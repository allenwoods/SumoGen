98  if (rng == 0) {
99  rng = &myRandomNumberGenerator;
100  }
101  // Polar method to avoid cosine
102  double u, q;
103  do {
104  u = rng->randExc(2.0) - 1;
105  const double v = rng->randExc(2.0) - 1;
106  q = u * u + v * v;
107  } while (q == 0.0 || q >= 1.0);
108  return (SUMOReal)(mean + variance * u * sqrt(-2 * log(q) / q));
109  }
