56 NBHelpers::relAngle(SUMOReal angle1, SUMOReal angle2) {
57  angle2 -= angle1;
58  if (angle2 > 180) {
59  angle2 = (360 - angle2) * -1;
60  }
61  while (angle2 < -180) {
62  angle2 = 360 + angle2;
63  }
64  return angle2;
65 }
