51 SUMOSAXAttributesImpl_Binary::SUMOSAXAttributesImpl_Binary(
52  const std::map<int, std::string>& predefinedTagsMML,
53  const std::string& objectType,
54  BinaryInputDevice* in) : SUMOSAXAttributes(objectType), myAttrIds(predefinedTagsMML) {
55  while (in->peek() == BinaryFormatter::BF_XML_ATTRIBUTE) {
56  unsigned char attr;
57  *in >> attr;
58  int type = in->peek();
59  switch (type) {
60  case BinaryFormatter::BF_BYTE:
61  *in >> myCharValues[attr];
62  break;
63  case BinaryFormatter::BF_INTEGER:
64  *in >> myIntValues[attr];
65  break;
66  case BinaryFormatter::BF_FLOAT:
67  case BinaryFormatter::BF_SCALED2INT:
68  *in >> myFloatValues[attr];
69  break;
70  case BinaryFormatter::BF_STRING:
71  *in >> myStringValues[attr];
72  break;
73  case BinaryFormatter::BF_LIST: {
74  int size;
75  *in >> size;
76  while (size > 0) {
77  const int type = in->peek();
78  if (type != BinaryFormatter::BF_POSITION_2D && type != BinaryFormatter::BF_POSITION_3D &&
79  type != BinaryFormatter::BF_SCALED2INT_POSITION_2D &&
80  type != BinaryFormatter::BF_SCALED2INT_POSITION_3D) {
81  throw ProcessError("Invalid binary file, only supporting position vectors.");
82  }
83  size--;
84  Position p;
85  *in >> p;
86  myPositionVectors[attr].push_back(p);
87  }
88  break;
89  }
90  case BinaryFormatter::BF_EDGE:
91  *in >> myIntValues[attr];
92  break;
93  case BinaryFormatter::BF_LANE:
94  *in >> myIntValues[attr];
95  in->putback(BinaryFormatter::BF_BYTE);
96  *in >> myCharValues[attr];
97  break;
98  case BinaryFormatter::BF_POSITION_2D:
99  case BinaryFormatter::BF_POSITION_3D:
100  case BinaryFormatter::BF_SCALED2INT_POSITION_2D:
101  case BinaryFormatter::BF_SCALED2INT_POSITION_3D: {
102  Position p;
103  *in >> p;
104  myPositionVectors[attr].push_back(p);
105  break;
106  }
107  case BinaryFormatter::BF_BOUNDARY: {
108  Position p;
109  *in >> p;
110  myPositionVectors[attr].push_back(p);
111  in->putback(BinaryFormatter::BF_POSITION_2D);
112  *in >> p;
113  myPositionVectors[attr].push_back(p);
114  break;
115  }
116  case BinaryFormatter::BF_COLOR:
117  *in >> myIntValues[attr];
118  break;
119  case BinaryFormatter::BF_NODE_TYPE:
120  *in >> myCharValues[attr];
121  break;
122  case BinaryFormatter::BF_EDGE_FUNCTION:
123  *in >> myCharValues[attr];
124  break;
125  case BinaryFormatter::BF_ROUTE: {
126  std::ostringstream into(std::ios::binary);
127  int size;
128  *in >> size;
129  FileHelpers::writeByte(into, BinaryFormatter::BF_ROUTE);
130  FileHelpers::writeInt(into, size);
131  if (size > 0) {
132  int intsToRead = size - 1;
133  int bitsOrEntry;
134  in->putback(BinaryFormatter::BF_INTEGER);
135  *in >> bitsOrEntry;
136  FileHelpers::writeInt(into, bitsOrEntry);
137  if (bitsOrEntry < 0) {
138  intsToRead = (-bitsOrEntry * (size - 1) - 1) / sizeof(int) / 8 + 2;
139  }
140  while (intsToRead > 0) {
141  in->putback(BinaryFormatter::BF_INTEGER);
142  *in >> bitsOrEntry;
143  FileHelpers::writeInt(into, bitsOrEntry);
144  intsToRead--;
145  }
146  }
147  myStringValues[attr] = into.str();
148  break;
149  }
150  default:
151  throw ProcessError("Invalid binary file");
152  }
153  myAttrs.insert(attr);
154  }
155 }
