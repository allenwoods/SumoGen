46 bool
47 TraCIServerAPI_MeMeDetector::processGet(TraCIServer& server, tcpip::Storage& inputStorage,
48  tcpip::Storage& outputStorage) {
49  // variable & id
50  int variable = inputStorage.readUnsignedByte();
51  std::string id = inputStorage.readString();
52  // check variable
53  if (variable != ID_LIST && variable != LAST_STEP_VEHICLE_NUMBER && variable != LAST_STEP_MEAN_SPEED
54  && variable != LAST_STEP_VEHICLE_ID_LIST && variable != LAST_STEP_VEHICLE_HALTING_NUMBER
55  && variable != ID_COUNT) {
56  return server.writeErrorStatusCmd(CMD_GET_MULTI_ENTRY_EXIT_DETECTOR_VARIABLE, "Get MeMeDetector Variable: unsupported variable " + toHex(variable, 2) + " specified", outputStorage);
57  }
58  // begin response building
59  tcpip::Storage tempMsg;
60  // response-code, variableID, objectID
61  tempMsg.writeUnsignedByte(RESPONSE_GET_MULTI_ENTRY_EXIT_DETECTOR_VARIABLE);
62  tempMsg.writeUnsignedByte(variable);
63  tempMsg.writeString(id);
64  if (variable == ID_LIST) {
65  std::vector<std::string> ids;
66  MSNet::getInstance()->getDetectorControl().getTypedDetectors(SUMO_TAG_ENTRY_EXIT_DETECTOR).insertIDs(ids);
67  tempMsg.writeUnsignedByte(TYPE_STRINGLIST);
68  tempMsg.writeStringList(ids);
69  } else if (variable == ID_COUNT) {
70  std::vector<std::string> ids;
71  MSNet::getInstance()->getDetectorControl().getTypedDetectors(SUMO_TAG_ENTRY_EXIT_DETECTOR).insertIDs(ids);
72  tempMsg.writeUnsignedByte(TYPE_INTEGER);
73  tempMsg.writeInt((int) ids.size());
74  } else {
75  MSE3Collector* e3 = static_cast<MSE3Collector*>(MSNet::getInstance()->getDetectorControl().getTypedDetectors(SUMO_TAG_ENTRY_EXIT_DETECTOR).get(id));
76  if (e3 == 0) {
77  return server.writeErrorStatusCmd(CMD_GET_MULTI_ENTRY_EXIT_DETECTOR_VARIABLE, "Areal detector '" + id + "' is not known", outputStorage);
78  }
79  switch (variable) {
80  case ID_LIST:
81  break;
82  case LAST_STEP_VEHICLE_NUMBER:
83  tempMsg.writeUnsignedByte(TYPE_INTEGER);
84  tempMsg.writeInt(e3->getVehiclesWithin());
85  break;
86  case LAST_STEP_MEAN_SPEED:
87  tempMsg.writeUnsignedByte(TYPE_DOUBLE);
88  tempMsg.writeDouble(e3->getCurrentMeanSpeed());
89  break;
90  case LAST_STEP_VEHICLE_ID_LIST: {
91  tempMsg.writeUnsignedByte(TYPE_STRINGLIST);
92  std::vector<std::string> ids = e3->getCurrentVehicleIDs();
93  tempMsg.writeStringList(ids);
94  }
95  break;
96  case LAST_STEP_VEHICLE_HALTING_NUMBER:
97  tempMsg.writeUnsignedByte(TYPE_INTEGER);
98  tempMsg.writeInt(e3->getCurrentHaltingNumber());
99  break;
100  default:
101  break;
102  }
103  }
104  server.writeStatusCmd(CMD_GET_MULTI_ENTRY_EXIT_DETECTOR_VARIABLE, RTYPE_OK, "", outputStorage);
105  server.writeResponseWithLength(outputStorage, tempMsg);
106  return true;
107 }
