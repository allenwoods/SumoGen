418 bool
419 NBFrame::checkOptions() {
420  OptionsCont& oc = OptionsCont::getOptions();
421  bool ok = true;
422  //
423  if (!oc.isDefault("tls-guess.joining")) {
424  WRITE_WARNING("'--tls-guess.joining' was joined with '--tls.join'.\n Please use '--tls.join' in future only.");
425  if (!oc.isSet("tls.join")) {
426  oc.set("tls.join", "true");
427  }
428  }
429  if (!SUMOXMLDefinitions::TrafficLightTypes.hasString(oc.getString("tls.default-type"))) {
430  WRITE_ERROR("unsupported value '" + oc.getString("tls.default-type") + "' for option '--tls.default-type'");
431  ok = false;
432  }
433  if (oc.isSet("keep-edges.in-boundary") && oc.isSet("keep-edges.in-geo-boundary")) {
434  WRITE_ERROR("only one of the options 'keep-edges.in-boundary' or 'keep-edges.in-geo-boundary' may be given");
435  ok = false;
436  }
437  if (oc.getBool("no-internal-links") && oc.getBool("crossings.guess")) {
438  WRITE_ERROR("only one of the options 'no-internal-links' or 'crossings.guess' may be given");
439  ok = false;
440  }
441  if (!oc.isDefault("tls.green.time") && !oc.isDefault("tls.cycle.time")) {
442  WRITE_ERROR("only one of the options 'tls.green.time' or 'tls.cycle.time' may be given");
443  ok = false;
444  }
445  if (oc.getInt("junctions.internal-link-detail") < 2) {
446  WRITE_ERROR("junctions.internal-link-detail must >= 2");
447  ok = false;
448  }
449  return ok;
450 }
