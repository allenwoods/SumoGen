149 long
150 GUIParameterTableWindow::onRightButtonPress(FXObject* sender,
151  FXSelector sel,
152  void* data) {
153  // check which value entry was pressed
154  myTable->onLeftBtnPress(sender, sel, data);
155  int row = myTable->getCurrentRow();
156  if (row == -1 || row >= (int)(myItems.size())) {
157  return 1;
158  }
159  GUIParameterTableItemInterface* i = myItems[row];
160  if (!i->dynamic()) {
161  return 1;
162  }
163  if (myObject == 0) {
164  return 1;
165  }
166 
167  GUIParam_PopupMenuInterface* p = new GUIParam_PopupMenuInterface(*myApplication, *this, *myObject, i->getName(), i->getSUMORealSourceCopy());
168  new FXMenuCommand(p, "Open in new Tracker", 0, p, MID_OPENTRACKER);
169  // set geometry
170  p->setX(static_cast<FXEvent*>(data)->root_x);
171  p->setY(static_cast<FXEvent*>(data)->root_y);
172  p->create();
173  // show
174  p->show();
175  return 1;
176 }
