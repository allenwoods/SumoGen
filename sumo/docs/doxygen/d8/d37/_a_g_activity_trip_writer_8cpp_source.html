63 AGActivityTripWriter::addTrip(const AGTrip& trip) {
64  int time = (trip.getDay() - 1) * 86400 + trip.getTime();
65 
66  myTripOutput.openTag(SUMO_TAG_TRIP)
67  .writeAttr(SUMO_ATTR_ID, trip.getVehicleName())
68  .writeAttr(SUMO_ATTR_TYPE, trip.getType())
69  .writeAttr(SUMO_ATTR_DEPART, time)
70  .writeAttr(SUMO_ATTR_DEPARTPOS, trip.getDep().getPosition())
71  .writeAttr(SUMO_ATTR_ARRIVALPOS, trip.getArr().getPosition())
72  .writeAttr(SUMO_ATTR_ARRIVALSPEED, 0.)
73  .writeAttr(SUMO_ATTR_FROM, trip.getDep().getStreet().getID());
74 
75  if (!trip.getPassed()->empty()) {
76  std::ostringstream oss;
77  for (std::list<AGPosition>::const_iterator it = trip.getPassed()->begin(); it != trip.getPassed()->end(); ++it) {
78  if (it != trip.getPassed()->begin()) {
79  oss << " ";
80  }
81  oss << it->getStreet().getID();
82  }
83  myTripOutput.writeAttr(SUMO_ATTR_VIA, oss.str());
84  }
85  myTripOutput.writeAttr(SUMO_ATTR_TO, trip.getArr().getStreet().getID());
86  myTripOutput.closeTag();
87 }
