374  #region CalcPower
375  public double CalcPower(double speed, double acc, double gradient)
376  {
377  //Declaration
378  double power = 0;
379  double rotFactor = GetRotationalCoeffecient(speed);
380  double powerAux = (_auxPower * _ratedPower);
381 
382  //Calculate the power
383  power += (_massVehicle + _vehicleLoading) * Constants.GRAVITY_CONST * (_resistanceF0 + _resistanceF1 * speed + _resistanceF4 * Math.Pow(speed, 4)) * speed;
384  power += (_crossSectionalArea * _cWValue * Constants.AIR_DENSITY_CONST / 2) * Math.Pow(speed, 3);
385  power += (_massVehicle * rotFactor + _vehicleMassRot + _vehicleLoading) * acc * speed;
386  power += (_massVehicle + _vehicleLoading) * Constants.GRAVITY_CONST * gradient * 0.01 * speed;
387  power /= 1000;
388  power /= Constants._DRIVE_TRAIN_EFFICIENCY;
389  power += powerAux;
390 
391  //Return result
392  return power;
393  }

669  #region FindLowerUpperInPattern
670  private void FindLowerUpperInPattern(out int lowerIndex, out int upperIndex, List<double> pattern, double value)
671  {
672  lowerIndex = 0;
673  upperIndex = 0;
674 
675  if (value <= pattern.First())
676  {
677  lowerIndex = 0;
678  upperIndex = 0;
679  return;
680  }
681 
682  if (value >= pattern.Last())
683  {
684  lowerIndex = pattern.Count - 1;
685  upperIndex = pattern.Count - 1;
686  return;
687  }
688 
689  // bisection search to find correct position in power pattern
690  int middleIndex = (pattern.Count - 1) / 2;
691  upperIndex = pattern.Count - 1;
692  lowerIndex = 0;
693 
694  while (upperIndex - lowerIndex > 1)
695  {
696  if (pattern[middleIndex] == value)
697  {
698  lowerIndex = middleIndex;
699  upperIndex = middleIndex;
700  return;
701  }
702  else if (pattern[middleIndex] < value)
703  {
704  lowerIndex = middleIndex;
705  middleIndex = (upperIndex - lowerIndex) / 2 + lowerIndex;
706  }
707  else
708  {
709  upperIndex = middleIndex;
710  middleIndex = (upperIndex - lowerIndex) / 2 + lowerIndex;
711  }
712  }
713 
714  if (pattern[lowerIndex] <= value && value < pattern[upperIndex])
715  {
716  return;
717  }
718  }
