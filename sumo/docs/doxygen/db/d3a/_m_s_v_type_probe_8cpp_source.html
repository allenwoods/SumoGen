66 MSVTypeProbe::execute(SUMOTime currentTime) {
67  myOutputDevice.openTag(SUMO_TAG_TIMESTEP);
68  myOutputDevice.writeAttr(SUMO_ATTR_TIME, time2string(currentTime));
69  myOutputDevice.writeAttr(SUMO_ATTR_ID, getID());
70  myOutputDevice.writeAttr("vType", myVType);
71  MSVehicleControl& vc = MSNet::getInstance()->getVehicleControl();
72  for (MSVehicleControl::constVehIt it = vc.loadedVehBegin(); it != vc.loadedVehEnd(); ++it) {
73  const SUMOVehicle* veh = it->second;
74  const MSVehicle* microVeh = dynamic_cast<const MSVehicle*>(veh);
75  if (myVType == "" || myVType == veh->getVehicleType().getID()) {
76  if (veh->isOnRoad()) {
77  Position pos = veh->getPosition();
78  myOutputDevice.openTag(SUMO_TAG_VEHICLE);
79  myOutputDevice.writeAttr(SUMO_ATTR_ID, veh->getID());
80  if (microVeh != 0) {
81  myOutputDevice.writeAttr(SUMO_ATTR_LANE, microVeh->getLane()->getID());
82  }
83  myOutputDevice.writeAttr(SUMO_ATTR_POSITION, veh->getPositionOnLane());
84  myOutputDevice.writeAttr(SUMO_ATTR_X, pos.x());
85  myOutputDevice.writeAttr(SUMO_ATTR_Y, pos.y());
86  if (MSNet::getInstance()->hasElevation()) {
87  myOutputDevice.writeAttr(SUMO_ATTR_Z, pos.z());
88  }
89  if (GeoConvHelper::getFinal().usingGeoProjection()) {
90  GeoConvHelper::getFinal().cartesian2geo(pos);
91  myOutputDevice.setPrecision(GEO_OUTPUT_ACCURACY);
92  myOutputDevice.writeAttr(SUMO_ATTR_LAT, pos.y());
93  myOutputDevice.writeAttr(SUMO_ATTR_LON, pos.x());
94  myOutputDevice.setPrecision();
95  }
96  myOutputDevice.writeAttr(SUMO_ATTR_SPEED, veh->getSpeed());
97  myOutputDevice.closeTag();
98  }
99  }
100  }
101  myOutputDevice.closeTag();
102  return myFrequency;
103 }
