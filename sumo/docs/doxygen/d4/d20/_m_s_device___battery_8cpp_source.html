118 bool MSDevice_Battery::notifyMove(SUMOVehicle& veh, SUMOReal /* oldPos */, SUMOReal /* newPos */, SUMOReal /* newSpeed */) {
119  // Start vehicleStoppedTimer if the vehicle is stopped. In other case reset timer
120  if (veh.getSpeed() < SUMO_const_haltingSpeed)
121  // Increase vehicle stopped timer
122  {
123  increaseVehicleStoppedTimer();
124  } else
125  // Reset vehicle Stopped
126  {
127  resetVehicleStoppedTimer();
128  }
129 
130  // Update Energy from the battery
131  if (getMaximumBatteryCapacity() != 0) {
132  myConsum = getPropEnergy(veh);
133 
134  // Energy lost/gained from vehicle movement (via vehicle energy model) [kWh]
135  setActualBatteryCapacity(getActualBatteryCapacity() - myConsum);
136 
137  // saturate between 0 and myMaximumBatteryCapacity [kWh]
138  if (getActualBatteryCapacity() < 0) {
139  setActualBatteryCapacity(0);
140 
141  // Show warning if battery is depleted
142  if (getMaximumBatteryCapacity() > 0)
143  WRITE_WARNING("Battery of vehicle '" + veh.getID() + "' is depleted.")
144 
145  } else if (getActualBatteryCapacity() > getMaximumBatteryCapacity()) {
146  setActualBatteryCapacity(getMaximumBatteryCapacity());
147  }
148 
149  setLastAngle(veh.getAngle());
150  }
151 
152  // Check if vehicle has under their position one charge Station
153  std::string ChargingStationID = MSNet::getInstance()->getChargingStationID(veh.getLane(), veh.getPositionOnLane());
154 
155  // If vehicle is over a charging station
156  if (ChargingStationID != "") {
157  // if the vehicle is almost stopped, or charge in transit is enabled, then charge vehicle
158  if ((veh.getSpeed() < SUMO_const_haltingSpeed) || (MSNet::getInstance()->getChargingStation(ChargingStationID)->getChargeInTransit() == 1)) {
159  // Set Flags Stopped/intransit to
160  if (veh.getSpeed() < SUMO_const_haltingSpeed) {
161  // vehicle ist almost stopped, then is charging stopped
162  myChargingStopped = true;
163 
164  // therefore isn't charging in transit
165  myChargingInTransit = false;
166  } else {
167  // vehicle is moving, and the Charging station allow charge in transit
168  myChargingStopped = false;
169 
170  // Therefore charge in transit
171  myChargingInTransit = true;
172  }
173 
174  // get pointer to charging station
175  myActChargingStation = MSNet::getInstance()->getChargingStation(ChargingStationID);
176 
177  // Only update charging start time if vehicle allow charge in transit, or in other case
178  // if the vehicle not allow charge in transit but it's stopped.
179  if (myActChargingStation->getChargeInTransit() == true || veh.getSpeed() < SUMO_const_haltingSpeed)
180  // Update Charging start time
181  {
182  increaseChargingStartTime();
183  }
184 
185  // time it takes the vehicle at the station < charging station time delay?
186  if (getChargingStartTime() > myActChargingStation->getChargeDelay()) {
187  // Enable charging vehicle
188  myActChargingStation->setChargingVehicle(true);
189 
190  // Calulate energy charged
191  myEnergyCharged = myActChargingStation->getChargingPower() * myActChargingStation->getEfficency();
192 
193  // Convert from [kWs] to [kWh] (3600s / 1h):
194  myEnergyCharged /= 3600;
195 
196  // Update Battery charge
197  if ((myEnergyCharged + getActualBatteryCapacity()) > getMaximumBatteryCapacity()) {
198  setActualBatteryCapacity(getMaximumBatteryCapacity());
199  } else {
200  setActualBatteryCapacity(getActualBatteryCapacity() + myEnergyCharged);
201  }
202  }
203  }
204  }
205  // In other case, vehicle will be not charged
206  else {
207  // Disable flags
208  myChargingInTransit = false;
209  myChargingStopped = false;
210 
211  // Disable charging vehicle
212  if (myActChargingStation != NULL) {
213  myActChargingStation->setChargingVehicle(false);
214  }
215 
216  // Set charging station pointer to NULL
217  myActChargingStation = NULL;
218 
219  // Set energy charged to 0
220  myEnergyCharged = 0.00;
221 
222  // Reset timer
223  resetChargingStartTime();
224  }
225 
226  // Always return true.
227  return true;
228 }

231 bool MSDevice_Battery::notifyEnter(SUMOVehicle& veh, MSMoveReminder::Notification reason) {
232  // Set last Energy ONLY when the vehicle is introduced in the simulation
233  if (reason == MSMoveReminder::NOTIFICATION_DEPARTED) {
234  setLastEnergy(getMass() * veh.getSpeed() * veh.getSpeed() / 2 + getMass() * 9.81 * veh.getLane()->getShape().front().z() + getInternalMomentOfInertia() * 0.5 * veh.getSpeed() * veh.getSpeed());
235  setLastAngle(veh.getLane()->getShape().beginEndAngle());
236  }
237 
238  // This function return always true
239  return true;
240 }

640 SUMOReal MSDevice_Battery::getPropEnergy(SUMOVehicle& veh) {
641  // calculate current kinetic energy
642  SUMOReal height_cur = veh.getPositionOnLane() / veh.getLane()->getLength() * (veh.getLane()->getShape().back().z() - veh.getLane()->getShape().front().z());
643 
644  // kinetic energy of vehicle with current velocity
645  SUMOReal currentEnergy = 0.5 * getMass() * veh.getSpeed() * veh.getSpeed();
646 
647  // add current potential energy of vehicle at current position
648  currentEnergy += getMass() * 9.81 * height_cur;
649 
650  // Calculate the radius of the vehicle's current path if is distintc (r = ds / dphi)
651  SUMOReal radius = 0;
652 
653  // If angle of vehicle was changed
654  if (getLastAngle() != veh.getAngle()) {
655  // Compute new radio
656  radius = SPEED2DIST(veh.getSpeed()) / fabs(GeomHelper::angleDiff(getLastAngle(), veh.getAngle()));
657 
658  // Check if radius is in the interval [0.0001 - 10000] (To avoid overflow and division by zero)
659  if (radius < 0.0001) {
660  radius = 0.0001;
661  } else if (radius > 10000) {
662  radius = 10000;
663  }
664  }
665 
666  // add current rotational energy of internal rotating elements
667  currentEnergy += getInternalMomentOfInertia() * veh.getSpeed() * veh.getSpeed();
668 
669  // kinetic + potential + rotational energy gain [Ws] (MODIFICATED LAST ANGLE)
670  SUMOReal EnergyLoss = (currentEnergy - getLastEnergy());
671 
672  // save current total energy for next time step
673  setLastEnergy(currentEnergy);
674 
675  // Calculate energy losses:
676  // EnergyLoss,Air = 1/2 * rho_air [kg/m^3] * myFrontSurfaceArea [m^2] * myAirDragCoefficient [-] * v_Veh^2 [m/s] * s [m]
677  // ... with rho_air [kg/m^3] = 1,2041 kg/m^3 (at T = 20C)
678  // ... with s [m] = v_Veh [m/s] * 1 [s]
679  EnergyLoss += 0.5 * 1.2041 * getFrontSurfaceArea() * getAirDragCoefficient() * fabs(veh.getSpeed() * veh.getSpeed() * veh.getSpeed());
680 
681  // Energy loss through Air resistance [Ws]
682  // EnergyLoss,Tire = c_R [-] * F_N [N] * s [m]
683  // ... with c_R = ~0.012 (car tire on asphalt)
684  // ... with F_N [N] = myMass [kg] * g [m/s^2]
685  EnergyLoss += getRollDragCoefficient() * 9.81 * getMass() * fabs(veh.getSpeed());
686 
687  // Energy loss through Roll resistance [Ws]
688  // ... (fabs(veh.getSpeed())>=0.01) = 0, if vehicle isn't moving
689  // EnergyLoss,internalFrictionRadialForce = c [m] * F_rad [N];
690  if (getLastAngle() != veh.getAngle()) {
691  // Energy loss through friction by radial force [Ws]
692  EnergyLoss += getRadialDragCoefficient() * getMass() * veh.getSpeed() * veh.getSpeed() / radius;
693  }
694 
695  // EnergyLoss,constantConsumers
696  // Energy loss through constant loads (e.g. A/C) [Ws]
697  EnergyLoss += getConstantPowerIntake();
698 
699  //E_Bat = E_kin_pot + EnergyLoss;
700  if (EnergyLoss > 0) {
701  // Assumption: Efficiency of myPropulsionEfficiency when accelerating
702  EnergyLoss = EnergyLoss / getPropulsionEfficiency();
703  } else {
704  // Assumption: Efficiency of myRecuperationEfficiency when recuperating
705  EnergyLoss = EnergyLoss * getRecuperationEfficiency();
706  }
707 
708  // convert from [Ws] to [kWh] (3600s / 1h):
709  EnergyLoss = EnergyLoss / 3600 ; // EnergyLoss[Ws] * 1[h]/3600[s] * 1[k]/1000
710 
711  // Return calculated energy
712  return EnergyLoss;
713 }
